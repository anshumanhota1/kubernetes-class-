# ————— All commands from the Kubernetes kubeadm + containerd setup —————

# 1) Disable Swap (all nodes)
sudo swapoff -a
sudo sed -i '/ swap / s/^/#/' /etc/fstab

# 2) Enable IPv4 forwarding and network params (all nodes)

cat <<EOF | sudo tee /etc/sysctl.d/k8s.conf
net.ipv4.ip_forward = 1
EOF
sudo sysctl --system

# 3) Load necessary kernel modules (all nodes)
cat <<EOF | sudo tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF
sudo modprobe overlay
sudo modprobe br_netfilter

# 4) Install containerd (all nodes)
sudo apt-get update
sudo apt-get install -y ca-certificates curl
sudo install -m 0755 -d /etc/apt/keyrings
sudo curl -fsSL https://download.docker.com/linux/ubuntu/gpg -o /etc/apt/keyrings/docker.asc
sudo chmod a+r /etc/apt/keyrings/docker.asc

sudo apt-get update
sudo apt-get install -y containerd

# 5) Configure containerd
sudo mkdir -p /etc/containerd
containerd config default | sudo tee /etc/containerd/config.toml
# (edit config.toml to set SystemdCgroup = true)
sudo sed -i 's/SystemdCgroup = false/SystemdCgroup = true/' /etc/containerd/config.toml
sudo systemctl restart containerd

# 6) Install kubeadm, kubelet, and kubectl (all nodes)
sudo apt-get update
sudo apt-get install -y apt-transport-https ca-certificates curl gpg
curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.30/deb/Release.key | sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg
echo "deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.30/deb/ /" | sudo tee /etc/apt/sources.list.d/kubernetes.list

sudo apt-get update
sudo apt-get install -y kubelet kubeadm kubectl
sudo apt-mark hold kubelet kubeadm kubectl

# Optional: enable kubelet service
sudo systemctl enable --now kubelet

# 7) Initialize Kubernetes control plane (on master/control plane node)
sudo kubeadm init --pod-network-cidr=192.168.0.0/16

# 8) Set up kubeconfig on the control plane node
mkdir -p $HOME/.kube
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
sudo chown $(id -u):$(id -g) $HOME/.kube/config

# 9) Deploy a CNI network addon (example: Calico)
kubectl apply -f https://raw.githubusercontent.com/projectcalico/calico/v3.25/manifests/tigera-operator.yaml
curl -LO https://raw.githubusercontent.com/projectcalico/calico/v3.25/manifests/custom-resources.yaml
kubectl apply -f custom-resources.yaml

# (Alternative CNI example: Flannel)
kubectl apply -f https://raw.githubusercontent.com/coreos/flannel/master/Documentation/kube-flannel.yml

# 10) Join worker nodes to the cluster
# This command is *generated by kubeadm init output*; e.g.:
sudo kubeadm join <control-plane-ip>:6443 --token <token> \
  --discovery-token-ca-cert-hash sha256:<hash>

# Other useful kubeadm commands
sudo kubeadm token create --print-join-command

# 11) Check cluster / nodes status
kubectl get nodes
kubectl get pods -A

# ————— End of commands —————



echo 'export PATH=$PATH:/snap/bin' >> ~/.bashrc
source ~/.bashrc
